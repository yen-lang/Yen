Language Features

  • Clean syntax — let / var for immutable/mutable bindings
  • Functions & lambdas — func name() {} and |x| x * 2
  • Pattern matching — match (val) { 1 => "one"; _ => "other" }
  • Pipe operator — data |> transform |> print
  • Go-style concurrency — go func() with channels
  • Try/catch/throw — structured error handling
  • Classes & structs — OOP with methods and fields
  • Destructuring — let [a, b, c] = list;
  • String interpolation — "Hello ${name}!"
  • Spread operator — [...a, ...b] and func(...args)

Standard Library

  • import 'net.http' — HTTP client & server
  • import 'net.socket' — TCP/UDP sockets
  • import 'regex' — Regular expressions
  • import 'os' — File system, process, environment
  • import 'async' — Channels, goroutines, sleep

Example

import 'net.http';
import 'async';

let server = http_server(8080);

func handle(server) {
    while (true) {
        let req = http_server_next(server);
        print "[${req['method']}] ${req['path']}";
        http_server_respond(req["client"], 200,
            {"Content-Type": "text/plain"},
            "Hello from Yen!");
    }
}

go handle(server);
print "Server running on :8080";
